<script>
    // ESTA ES LA MAGIA QUE HACE QUE FUNCIONE
    function goToScreen(screenId) {
        // Oculta todas las pantallas
        const screens = document.querySelectorAll('.screen');
        screens.forEach(s => s.classList.remove('active'));
        
        // Muestra la pantalla seleccionada
        document.getElementById(screenId).classList.add('active');
    }

    function seleccionarHora(elemento) {
        // Quita el color de otros
        document.querySelectorAll('.machine-card').forEach(c => c.style.border = "none");
        // Marca el seleccionado
        elemento.style.border = "2px solid #1A6B5A";
        alert("Horario seleccionado");
    }

    function finalizar() {
        alert("¡Listo! Tu lavadora ha sido reservada.");
        goToScreen('screen-home');
    }
</script>
